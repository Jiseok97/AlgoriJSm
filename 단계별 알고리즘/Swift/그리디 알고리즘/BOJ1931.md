import Foundation

func greedy(_ arr: [[Int]]) -> Int {
    var start : Int = 0
    var result : Int = 0
    for t in arr {
        if t[0] >= start {
            start = t[1]
            result += 1
        }
    }
    return result
}

let cnt = Int(readLine()!)!
var meet = [[Int]]()
for _ in 0..<cnt {
    meet.append(readLine()!.split(separator: " ").map{Int($0)!})
}
meet.sort{ (a,b) -> Bool in
    if a[1] == b[1] {
        return a[0] < b[0]
    } else {
        return a[1] < b[1]
    }
}
print(greedy(meet))
