import heapq as hq
import sys
input = sys.stdin.readline

n = int(input())
min_h = []
ans = 0

for _ in range(n):
    hq.heappush(min_h, int(input()))


while len(min_h) > 1:
    a, b = hq.heappop(min_h), hq.heappop(min_h)
    ans += a + b
    hq.heappush(min_h, a + b)
    
print(ans)
